def each(&block)
  return to_enum unless block

  i = 0
  while i < @enums.size
    @pos = i
    @enums[i].each(&block)
    i += 1
  end

  self
end

def rewind
  while 0 <= @pos && @pos < @enums.size
    e = @enums[@pos]
    e.rewind if e.respond_to?(:rewind)
    @pos -= 1
  end

  self
end

def size
  @enums.reduce(0) do |a, e|
    return nil unless e.respond_to?(:size)
    a + e.size
  end
end
